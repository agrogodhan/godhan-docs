/*==========================================================
  CATTLE SMART COLLAR — MAIN FIRMWARE
  Device  : ESP32-S3
  Version : 1.0.0
  Flow    : BLE Provision (once) → WiFi → MQTT Cloud
==========================================================*/

#include <Arduino.h>
#include "ble_provision.h"
#include "wifi_manager.h"
#include "sensors.h"
#include "cloud.h"
#include "power.h"
#include <Preferences.h>       // NVS flash storage
#include <ArduinoJson.h>       // JSON payload builder

// ── Device Identity ──────────────────────────────────────
#define FIRMWARE_VERSION    "1.0.0"
#define DEVICE_PREFIX       "COLLAR-"
#define SEND_INTERVAL_MS    30000     // 30 seconds
#define ALERT_INTERVAL_MS   5000      // 5 sec on alert condition

// ── Globals ───────────────────────────────────────────────
Preferences   prefs;
SensorData    sensorData;
String        deviceId;
bool          provisioned  = false;
unsigned long lastSendTime = 0;

// ── Forward declarations ──────────────────────────────────
void buildAndSendPayload();
bool alertConditionMet();
String getDeviceId();

// ════════════════════════════════════════════════════════
  SETUP
════════════════════════════════════════════════════════
void setup() {
  Serial.begin(115200);
  Serial.println("\n[BOOT] Cattle Collar Firmware v" + String(FIRMWARE_VERSION));

  // 1. Get unique device ID from MAC
  deviceId = getDeviceId();
  Serial.println("[BOOT] Device ID: " + deviceId);

  // 2. Open NVS (persistent storage)
  prefs.begin("collar", false);

  // 3. Check if WiFi credentials exist in NVS
  provisioned = prefs.getBool("provisioned", false);

  if (!provisioned) {
    // ── FIRST BOOT: start BLE provisioning mode ──
    Serial.println("[BLE] No credentials found. Starting BLE provision...");
    startBLEProvisioning(deviceId, &prefs);
    // Blocks until phone sends WiFi credentials via BLE
    provisioned = true;
    Serial.println("[BLE] Credentials saved. BLE stopping...");
    stopBLE();   // Turn off BLE to save power permanently
  } else {
    Serial.println("[BOOT] Credentials found. Skipping BLE.");
  }

  // 4. Connect to WiFi using saved credentials
  String ssid = prefs.getString("ssid", "");
  String pass = prefs.getString("pass", "");
  wifiConnect(ssid, pass);

  // 5. Connect to MQTT cloud broker
  cloudConnect(deviceId);

  // 6. Initialise all sensors
  initSensors();

  // 7. Register device on cloud (first packet)
  cloudRegisterDevice(deviceId, FIRMWARE_VERSION);

  Serial.println("[BOOT] Setup complete. Entering main loop.");
}

// ════════════════════════════════════════════════════════
  MAIN LOOP
════════════════════════════════════════════════════════
void loop() {
  unsigned long now = millis();

  // Keep MQTT connection alive
  cloudLoop();

  // Reconnect WiFi if dropped
  if (!wifiConnected()) {
    Serial.println("[WiFi] Disconnected. Reconnecting...");
    wifiReconnect();
  }

  // Determine send interval (faster if alert condition)
  unsigned long interval = alertConditionMet()
                            ? ALERT_INTERVAL_MS
                            : SEND_INTERVAL_MS;

  if (now - lastSendTime >= interval) {
    lastSendTime = now;

    // Read all sensors
    readAllSensors(&sensorData);

    // Build JSON payload and publish to cloud
    buildAndSendPayload();

    // Log to serial
    Serial.printf("[DATA] Temp:%.1f HR:%d SpO2:%d%% Steps:%d\n",
      sensorData.bodyTempC, sensorData.heartRate,
      sensorData.spO2, sensorData.steps);
  }

  // Power management: light sleep between reads
  lightSleepUntilNextRead(interval, lastSendTime);
}

// ── Build JSON and send ───────────────────────────────────
void buildAndSendPayload() {
  StaticJsonDocument<512> doc;

  // Identity
  doc["device_id"]   = deviceId;
  doc["cow_id"]      = prefs.getString("cow_id", "UNSET");
  doc["timestamp"]   = getISOTimestamp();

  // Activity
  doc["steps"]            = sensorData.steps;
  doc["posture"]          = sensorData.posture;
  doc["activity_score"]   = sensorData.activityScore;

  // Health
  doc["body_temp_c"]        = sensorData.bodyTempC;
  doc["heart_rate_bpm"]     = sensorData.heartRate;
  doc["spo2_pct"]           = sensorData.spO2;
  doc["rumination_cycles"]  = sensorData.ruminationCycles;

  // Reproduction
  doc["vaginal_temp_c"]  = sensorData.vaginalTempC;
  doc["estrus_flag"]     = sensorData.estrusFlag;
  doc["calving_alert"]   = sensorData.calvingAlert;

  // Device status
  doc["battery_pct"]  = getBatteryPercent();
  doc["wifi_rssi"]    = getWiFiRSSI();
  doc["zone"]         = detectZone();    // "shed" or "open"

  String payload;
  serializeJson(doc, payload);
  cloudPublish("cattle/data", payload);
}

// ── Alert condition check ─────────────────────────────────
bool alertConditionMet() {
  return (sensorData.bodyTempC > 39.5f)   // Fever
      || (sensorData.heartRate  > 100)    // Tachycardia
      || (sensorData.spO2       < 92)     // Low oxygen
      || sensorData.estrusFlag             // Heat detected
      || sensorData.calvingAlert;          // Calving imminent
}

// ── Unique device ID from MAC ─────────────────────────────
String getDeviceId() {
  uint8_t mac[6];
  esp_read_mac(mac, ESP_MAC_WIFI_STA);
  char id[16];
  snprintf(id, sizeof(id), "%s%02X%02X%02X",
           DEVICE_PREFIX, mac[3], mac[4], mac[5]);
  return String(id);
}

/*==========================================================
  BLE PROVISIONING MODULE
  Purpose : Accept WiFi credentials + Cow ID from farmer app
  Method  : BLE GATT server with writable characteristics
  Runs    : ONCE on first boot, then BLE is permanently OFF
==========================================================*/

#pragma once
#include <BLEDevice.h>
#include <BLEServer.h>
#include <BLEUtils.h>
#include <BLE2902.h>
#include <Preferences.h>

// ── BLE Service + Characteristic UUIDs ───────────────────
#define PROVISION_SERVICE_UUID  "12345678-1234-1234-1234-123456789000"
#define SSID_CHAR_UUID          "12345678-1234-1234-1234-123456789001"
#define PASS_CHAR_UUID          "12345678-1234-1234-1234-123456789002"
#define COWID_CHAR_UUID         "12345678-1234-1234-1234-123456789003"
#define STATUS_CHAR_UUID        "12345678-1234-1234-1234-123456789004"
#define BLE_TIMEOUT_MS          300000   // 5 min timeout

static bool         bleProvDone   = false;
static String       rcvSSID, rcvPass, rcvCowId;
static BLEServer*   pServer       = nullptr;
static BLECharacteristic* pStatus = nullptr;

// ── Callback: handles writes from farmer's phone ──────────
class ProvisionCallback : public BLECharacteristicCallbacks {
  Preferences* _prefs;
public:
  ProvisionCallback(Preferences* p) : _prefs(p) {}

  void onWrite(BLECharacteristic* pChar) override {
    String uuid = pChar->getUUID().toString().c_str();
    String val  = pChar->getValue().c_str();

    if (uuid == SSID_CHAR_UUID)  { rcvSSID  = val; Serial.println("[BLE] SSID received");  }
    if (uuid == PASS_CHAR_UUID)  { rcvPass  = val; Serial.println("[BLE] Pass received");  }
    if (uuid == COWID_CHAR_UUID) { rcvCowId = val; Serial.println("[BLE] CowID received"); }

    // All 3 values received → save to NVS and signal done
    if (rcvSSID.length() > 0 &&
        rcvPass.length() > 0 &&
        rcvCowId.length() > 0) {

      _prefs->putString("ssid",   rcvSSID);
      _prefs->putString("pass",   rcvPass);
      _prefs->putString("cow_id", rcvCowId);
      _prefs->putBool  ("provisioned", true);

      // Notify phone: provisioning success
      pStatus->setValue("OK:PROVISIONED");
      pStatus->notify();

      Serial.println("[BLE] Provisioning complete! SSID=" + rcvSSID);
      bleProvDone = true;
    }
  }
};

// ── Start BLE provisioning server ────────────────────────
void startBLEProvisioning(String deviceId, Preferences* prefs) {
  BLEDevice::init(deviceId.c_str());
  pServer = BLEDevice::createServer();

  BLEService* svc = pServer->createService(PROVISION_SERVICE_UUID);

  auto makeChar = [&](const char* uuid) {
    return svc->createCharacteristic(uuid,
      BLECharacteristic::PROPERTY_WRITE |
      BLECharacteristic::PROPERTY_READ);
  };

  auto* cSSID  = makeChar(SSID_CHAR_UUID);
  auto* cPass  = makeChar(PASS_CHAR_UUID);
  auto* cCowId = makeChar(COWID_CHAR_UUID);
  pStatus = svc->createCharacteristic(STATUS_CHAR_UUID,
    BLECharacteristic::PROPERTY_NOTIFY |
    BLECharacteristic::PROPERTY_READ);
  pStatus->addDescriptor(new BLE2902());

  auto* cb = new ProvisionCallback(prefs);
  cSSID ->setCallbacks(cb);
  cPass ->setCallbacks(cb);
  cCowId->setCallbacks(cb);

  svc->start();
  BLEDevice::startAdvertising();
  Serial.println("[BLE] Advertising as: " + deviceId);

  // Block until provisioned or timeout
  unsigned long start = millis();
  while (!bleProvDone && (millis() - start < BLE_TIMEOUT_MS)) {
    delay(100);
  }
  if (!bleProvDone) Serial.println("[BLE] Timeout. Will retry on reboot.");
}

void stopBLE() {
  BLEDevice::stopAdvertising();
  BLEDevice::deinit(true);
  Serial.println("[BLE] BLE stack de-initialised. Radio OFF.");
}

/*==========================================================
  WiFi MANAGER MODULE
  Handles: connect, reconnect, RSSI zone detection
  Zone detection: "shed" if RSSI > -65dBm, else "open"
==========================================================*/

#pragma once
#include <WiFi.h>
#include <time.h>

#define WIFI_RETRY_LIMIT   20      // 10 seconds max
#define WIFI_RETRY_DELAY   500     // ms between retries
#define SHED_RSSI_THRESH   -65     // dBm: stronger = shed
#define NTP_SERVER         "pool.ntp.org"
#define TZ_OFFSET          19800   // IST = UTC+5:30

static String _ssid, _pass;

// ── Connect to WiFi ───────────────────────────────────────
void wifiConnect(String ssid, String pass) {
  _ssid = ssid; _pass = pass;
  WiFi::mode(WIFI_STA);
  WiFi::begin(ssid.c_str(), pass.c_str());

  Serial.print("[WiFi] Connecting to: " + ssid);
  int retries = 0;
  while (WiFi::status() != WL_CONNECTED && retries++ < WIFI_RETRY_LIMIT) {
    delay(WIFI_RETRY_DELAY);
    Serial.print(".");
  }

  if (WiFi::status() == WL_CONNECTED) {
    Serial.println("\n[WiFi] Connected! IP: " + WiFi::localIP().toString());
    Serial.println("[WiFi] RSSI: " + String(WiFi::RSSI()) + " dBm");
    syncNTP();
  } else {
    Serial.println("\n[WiFi] FAILED. Will retry in loop.");
  }
}

void wifiReconnect() { wifiConnect(_ssid, _pass); }

bool wifiConnected() { return WiFi::status() == WL_CONNECTED; }

int  getWiFiRSSI()   { return WiFi::RSSI(); }

// ── Zone Detection via RSSI ───────────────────────────────
// Shed APs are indoor → stronger signal → RSSI closer to 0
// Open area APs are farther → weaker → RSSI more negative
String detectZone() {
  return (getWiFiRSSI() >= SHED_RSSI_THRESH) ? "shed" : "open";
}

// ── Sync time from NTP ────────────────────────────────────
void syncNTP() {
  configTime(TZ_OFFSET, 0, NTP_SERVER);
  Serial.print("[NTP] Syncing time");
  struct tm ti;
  for (int i = 0; i < 20; i++) {
    if (getLocalTime(&ti)) { Serial.println(" OK"); return; }
    delay(500); Serial.print(".");
  }
  Serial.println(" FAILED");
}

// ── ISO 8601 timestamp ────────────────────────────────────
String getISOTimestamp() {
  struct tm ti;
  if (!getLocalTime(&ti)) return "1970-01-01T00:00:00Z";
  char buf[30];
  strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &ti);
  return String(buf);
}

/*==========================================================
  SENSOR DRIVER MODULE
  Sensors : MPU-6050 · MAX30102 · DS18B20 · INMP441 · LMT70
  All results written into SensorData struct
==========================================================*/

#pragma once
#include <Wire.h>
#include <MPU6050.h>
#include <MAX30105.h>
#include <heartRate.h>
#include <OneWire.h>
#include <DallasTemperature.h>
#include <driver/i2s.h>

// ── GPIO Pin Map ──────────────────────────────────────────
#define PIN_I2C_SDA       4
#define PIN_I2C_SCL       5
#define PIN_ONE_WIRE      18
#define PIN_I2S_WS        6
#define PIN_I2S_SCK       7
#define PIN_I2S_SD        8
#define PIN_ADC_LMT70     9

// ── Health thresholds (cattle normal ranges) ──────────────
#define TEMP_FEVER_C      39.5f
#define TEMP_ESTRUS_RISE  0.5f    // rise vs baseline = estrus
#define TEMP_CALVING_DROP 0.3f    // drop 12-24h before calving
#define HR_MAX            100
#define SPO2_MIN          92
#define RUM_LOW_THRESH    300     // jaw cycles/hr below = ill
#define I2S_SAMPLE_RATE   16000
#define RUM_SAMPLE_MS     5000    // 5 sec mic window

// ── Sensor Data Struct ────────────────────────────────────
struct SensorData {
  // Activity
  int     steps;
  String  posture;          // "grazing"|"standing"|"lying"|"walking"
  int     activityScore;    // 0–100
  // Health
  float   bodyTempC;
  int     heartRate;
  int     spO2;
  int     ruminationCycles;
  // Reproduction
  float   vaginalTempC;
  bool    estrusFlag;
  bool    calvingAlert;
};

// ── Sensor instances ──────────────────────────────────────
static MPU6050          mpu;
static MAX30105         hrSensor;
static OneWire          ow(PIN_ONE_WIRE);
static DallasTemperature ds(&ow);
static float            vagTempBaseline = 38.5f;

// ════════════════════════════════════════════════
  INIT ALL SENSORS
════════════════════════════════════════════════
void initSensors() {
  Wire::begin(PIN_I2C_SDA, PIN_I2C_SCL);

  // MPU-6050
  mpu.initialize();
  if (mpu.testConnection())
    Serial.println("[SENSOR] MPU-6050 OK");
  mpu.setFullScaleAccelRange(MPU6050_ACCEL_FS_4);  // ±4g
  mpu.setFullScaleGyroRange(MPU6050_GYRO_FS_500);   // ±500°/s
  mpu.setDLPFMode(MPU6050_DLPF_BW_20);              // low-pass filter

  // MAX30102
  if (hrSensor.begin(Wire, I2C_SPEED_FAST)) {
    hrSensor.setup(60, 4, 2, 400, 411, 4096);
    Serial.println("[SENSOR] MAX30102 OK");
  }

  // DS18B20
  ds.begin();
  ds.setResolution(12);   // 0.0625°C resolution
  Serial.println("[SENSOR] DS18B20 OK - found: "
    + String(ds.getDeviceCount()) + " device(s)");

  // INMP441 I2S
  i2s_config_t i2sCfg = {
    .mode            = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX),
    .sample_rate     = I2S_SAMPLE_RATE,
    .bits_per_sample = I2S_BITS_PER_SAMPLE_32BIT,
    .channel_format  = I2S_CHANNEL_FMT_ONLY_LEFT,
    .communication_format = I2S_COMM_FORMAT_I2S,
    .dma_buf_count   = 4,
    .dma_buf_len     = 512,
    .use_apll        = false
  };
  i2s_pin_config_t pinCfg = {
    .bck_io_num   = PIN_I2S_SCK,
    .ws_io_num    = PIN_I2S_WS,
    .data_out_num = I2S_PIN_NO_CHANGE,
    .data_in_num  = PIN_I2S_SD
  };
  i2s_driver_install(I2S_NUM_0, &i2sCfg, 0, nullptr);
  i2s_set_pin(I2S_NUM_0, &pinCfg);
  Serial.println("[SENSOR] INMP441 I2S OK");

  Serial.println("[SENSOR] LMT70 ADC on GPIO" + String(PIN_ADC_LMT70));
  analogSetAttenuation(ADC_11db);   // 0–3.3V range
}

// ════════════════════════════════════════════════
  READ ALL SENSORS
════════════════════════════════════════════════
void readAllSensors(SensorData* d) {
  readIMU(d);
  readHeartRate(d);
  readBodyTemp(d);
  readRumination(d);
  readVaginalTemp(d);
}

// ── MPU-6050: Activity, posture, steps ───────────────────
void readIMU(SensorData* d) {
  int16_t ax, ay, az, gx, gy, gz;
  mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);

  float accX = ax / 8192.0f;   // convert to g
  float accY = ay / 8192.0f;
  float accZ = az / 8192.0f;
  float magnitude = sqrt(accX*accX + accY*accY + accZ*accZ);

  // Step detection: threshold crossing
  static int   stepCount  = 0;
  static bool  stepHigh   = false;
  if (magnitude > 1.15f && !stepHigh) { stepCount++; stepHigh = true;  }
  if (magnitude < 1.05f)             { stepHigh = false; }
  d->steps = stepCount;

  // Posture: use Z-axis tilt (head angle)
  float tilt = atan2(accX, accZ) * 180.0f / PI;
  if      (magnitude > 1.3f)          d->posture = "walking";
  else if (tilt < -30.0f)              d->posture = "grazing";
  else if (abs(accZ) < 0.3f)           d->posture = "lying";
  else                                  d->posture = "standing";

  // Activity score 0–100 based on recent magnitude
  d->activityScore = (int)constrain((magnitude - 1.0f) * 200, 0, 100);
}

// ── MAX30102: Heart rate + SpO₂ ───────────────────────────
void readHeartRate(SensorData* d) {
  const int SAMPLES = 100;
  long irBuffer[100], redBuffer[100];
  for (int i = 0; i < SAMPLES; i++) {
    while (!hrSensor.available()) hrSensor.check();
    redBuffer[i] = hrSensor.getRed();
    irBuffer[i]  = hrSensor.getIR();
    hrSensor.nextSample();
  }
  int32_t spo2;
  int8_t  spo2Valid, hrValid;
  int32_t hr;
  maxim_heart_rate_and_oxygen_saturation(
    irBuffer, SAMPLES, redBuffer,
    &spo2, &spo2Valid, &hr, &hrValid);

  d->heartRate = hrValid  ? (int)hr   : 0;
  d->spO2      = spo2Valid ? (int)spo2 : 0;
}

// ── DS18B20: Body temperature ─────────────────────────────
void readBodyTemp(SensorData* d) {
  ds.requestTemperatures();
  float t = ds.getTempCByIndex(0);
  d->bodyTempC = (t != DEVICE_DISCONNECTED_C) ? t : 0.0f;
}

// ── INMP441: Rumination via jaw vibration counts ──────────
void readRumination(SensorData* d) {
  const int   BUF_LEN    = 512;
  int32_t     samples[512];
  size_t      bytesRead  = 0;
  int         peakCount  = 0;
  unsigned long t0       = millis();

  while (millis() - t0 < RUM_SAMPLE_MS) {
    i2s_read(I2S_NUM_0, samples, BUF_LEN * sizeof(int32_t), &bytesRead, 100);
    int n = bytesRead / sizeof(int32_t);
    bool inPeak = false;
    for (int i = 0; i < n; i++) {
      float v = abs(samples[i] / (float)0x7FFFFFFF);
      if (v > 0.02f && !inPeak) { peakCount++; inPeak = true; }
      if (v < 0.01f)             { inPeak = false; }
    }
  }
  // Scale 5-sec count to cycles per hour
  d->ruminationCycles = peakCount * (3600 / (RUM_SAMPLE_MS / 1000));
}

// ── LMT70: Vaginal temp (estrus + calving detection) ──────
void readVaginalTemp(SensorData* d) {
  // LMT70: Vout = 1.8639 - 0.01158 * T_celsius
  // → T = (1.8639 - Vout) / 0.01158
  int   raw    = analogRead(PIN_ADC_LMT70);
  float vout   = raw * (3.3f / 4095.0f);
  float temp   = (1.8639f - vout) / 0.01158f;
  d->vaginalTempC = temp;

  float delta = temp - vagTempBaseline;

  // Estrus: temp rises ≥ 0.5°C above baseline
  d->estrusFlag = (delta >= TEMP_ESTRUS_RISE);

  // Calving: temp drops ≥ 0.3°C below baseline
  d->calvingAlert = (delta <= -TEMP_CALVING_DROP);

  // Update baseline slowly (drift tracking)
  vagTempBaseline = vagTempBaseline * 0.995f + temp * 0.005f;
}

/*==========================================================
  CLOUD MQTT MODULE
  Broker  : AWS IoT Core (or any MQTT broker)
  Protocol: MQTT over TLS port 8883
  Topics  : cattle/data · cattle/alert · cattle/cmd
==========================================================*/

#pragma once
#include <WiFiClientSecure.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>

// ── MQTT Broker Config ────────────────────────────────────
#define MQTT_HOST     "your-broker.iot.amazonaws.com"
#define MQTT_PORT     8883
#define MQTT_QOS      1

// ── MQTT Topic Templates ──────────────────────────────────
#define TOPIC_DATA    "cattle/{id}/data"
#define TOPIC_ALERT   "cattle/{id}/alert"
#define TOPIC_CMD     "cattle/{id}/cmd"
#define TOPIC_REGISTER "cattle/register"

// ── TLS Certificates (paste from AWS IoT console) ─────────
const char ROOT_CA[] PROGMEM = R"EOF(
-----BEGIN CERTIFICATE-----
<Paste AWS Root CA cert here>
-----END CERTIFICATE-----
)EOF";

const char DEVICE_CERT[] PROGMEM = R"EOF(
-----BEGIN CERTIFICATE-----
<Paste device certificate here>
-----END CERTIFICATE-----
)EOF";

const char PRIVATE_KEY[] PROGMEM = R"EOF(
-----BEGIN RSA PRIVATE KEY-----
<Paste private key here>
-----END RSA PRIVATE KEY-----
)EOF";

static WiFiClientSecure tlsClient;
static PubSubClient    mqttClient(tlsClient);
static String           _deviceId;

// ── Build topic string ────────────────────────────────────
String topic(const char* tmpl) {
  String t = tmpl;
  t.replace("{id}", _deviceId);
  return t;
}

// ── Command handler (from cloud → collar) ─────────────────
void onMQTTMessage(char* top, byte* payload, unsigned int len) {
  String msg = String((char*)payload).substring(0, len);
  Serial.println("[MQTT] CMD received: " + msg);
  StaticJsonDocument<128> cmd;
  if (deserializeJson(cmd, msg) == DeserializationError::Ok) {
    if (cmd["action"] == "update_cow_id") {
      // Handle cloud-pushed cow ID update
      Serial.println("[CMD] CowID updated");
    }
  }
}

// ── Connect to MQTT broker ────────────────────────────────
void cloudConnect(String deviceId) {
  _deviceId = deviceId;
  tlsClient.setCACert(ROOT_CA);
  tlsClient.setCertificate(DEVICE_CERT);
  tlsClient.setPrivateKey(PRIVATE_KEY);

  mqttClient.setServer(MQTT_HOST, MQTT_PORT);
  mqttClient.setCallback(onMQTTMessage);
  mqttClient.setBufferSize(1024);

  Serial.print("[MQTT] Connecting...");
  while (!mqttClient.connected()) {
    if (mqttClient.connect(deviceId.c_str())) {
      mqttClient.subscribe(topic(TOPIC_CMD).c_str(), MQTT_QOS);
      Serial.println(" connected!");
    } else {
      Serial.print("."); delay(2000);
    }
  }
}

// ── Register device on cloud at first boot ────────────────
void cloudRegisterDevice(String devId, const char* fwVer) {
  StaticJsonDocument<128> doc;
  doc["device_id"]       = devId;
  doc["firmware_version"] = fwVer;
  doc["event"]            = "boot";
  String payload;
  serializeJson(doc, payload);
  mqttClient.publish(TOPIC_REGISTER, payload.c_str(), true);
}

// ── Publish sensor data ───────────────────────────────────
void cloudPublish(const char* topicKey, String& payload) {
  if (!mqttClient.connected()) cloudConnect(_deviceId);
  bool ok = mqttClient.publish(
    topic(topicKey).c_str(),
    payload.c_str(),
    false
  );
  Serial.println(ok ? "[MQTT] Sent OK" : "[MQTT] Send FAILED");
}

/*==========================================================
  POWER MANAGEMENT MODULE
  Strategy: Light sleep between sensor reads
            Deep sleep if battery critical (<5%)
            Wake on timer every 30 seconds
==========================================================*/

#pragma once
#include <esp_sleep.h>
#include <esp_wifi.h>

#define PIN_BATT_ADC      10      // Voltage divider from LiPo
#define BATT_DIVIDER      2.0f    // R1=R2=100kΩ → Vout = Vbatt/2
#define BATT_FULL_V       4.2f    // LiPo full voltage
#define BATT_EMPTY_V      3.0f    // LiPo cutoff voltage
#define BATT_CRITICAL_PCT 5       // Enter deep sleep below 5%

// ── Read battery percentage ───────────────────────────────
int getBatteryPercent() {
  int   raw   = analogRead(PIN_BATT_ADC);
  float vADC  = raw * (3.3f / 4095.0f);
  float vBatt = vADC * BATT_DIVIDER;
  int   pct   = (int)(((vBatt - BATT_EMPTY_V)
                  / (BATT_FULL_V - BATT_EMPTY_V)) * 100.0f);
  return constrain(pct, 0, 100);
}

// ── Light sleep between reads ─────────────────────────────
// WiFi stays connected during light sleep (modem sleep)
void lightSleepUntilNextRead(unsigned long interval,
                              unsigned long lastSend) {
  unsigned long elapsed = millis() - lastSend;
  if (elapsed >= interval) return;   // Already time to send

  uint64_t sleepUs = (interval - elapsed) * 1000ULL;

  // Put WiFi modem in sleep but keep TCP/MQTT alive
  esp_wifi_set_ps(WIFI_PS_MIN_MODEM);

  esp_sleep_enable_timer_wakeup(sleepUs);
  esp_light_sleep_start();   // CPU halts, WiFi stays

  esp_wifi_set_ps(WIFI_PS_NONE);  // Restore full WiFi on wake
}

// ── Deep sleep if battery critical ───────────────────────
void checkBatteryCritical() {
  if (getBatteryPercent() <= BATT_CRITICAL_PCT) {
    Serial.println("[PWR] Battery critical! Deep sleep 1hr.");
    esp_sleep_enable_timer_wakeup(3600ULL * 1000000ULL);
    esp_deep_sleep_start();   // Full shutdown, wakes after 1hr
  }
}

// ── Power mode summary ────────────────────────────────────
// ACTIVE   : ~240mA  → reading sensors + sending MQTT
// LIGHT SLP: ~15mA   → between reads, WiFi modem sleep
// DEEP SLP : ~0.01mA → battery critical, full shutdown
// Solar    : ~100mA  → net positive during daylight
